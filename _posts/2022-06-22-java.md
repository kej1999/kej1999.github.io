# for문과 if문에서 정의한 변수 소멸 시점

## if문 안에서 정의한 것은 if문이 끝나면 소멸됨
## for문안에서 정의한 것은 for문이 끝나면 소멸됨

public class VariableScopeExample {
	public static void main(String[] args) {
		int var1;
		
		if(true) {
			int var2;
			var1 = 10;
			var2 = 20;
		}
		
		var1 = 10;
		//var2 = 20; 
		
		for (int i = 0; i < 1; i++) {
			int var3;
			var1 = 10;
			var3 = 30;
		}

		var1 = 10;
		//var3 = 30;
	}
}
